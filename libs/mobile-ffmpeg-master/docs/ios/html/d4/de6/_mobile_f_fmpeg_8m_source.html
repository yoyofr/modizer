61 + (int)executeWithId:(long)newExecutionId andArguments:(NSArray*)arguments {
62  lastCommandOutput = [[NSMutableString alloc] init];
63 
64  FFmpegExecution* currentFFmpegExecution = [[FFmpegExecution alloc] initWithExecutionId:newExecutionId andArguments:arguments];
65  [executionsLock lock];
66  [executions addObject: currentFFmpegExecution];
67  [executionsLock unlock];
68 
69  char **commandCharPArray = (char **)av_malloc(sizeof(char*) * ([arguments count] + 1));
70 
71  /* PRESERVING CALLING FORMAT
72  *
73  * ffmpeg <arguments>
74  */
75  commandCharPArray[0] = (char *)av_malloc(sizeof(char) * ([LIB_NAME length] + 1));
76  strcpy(commandCharPArray[0], [LIB_NAME UTF8String]);
77 
78  for (int i=0; i < [arguments count]; i++) {
79  NSString *argument = [arguments objectAtIndex:i];
80  commandCharPArray[i + 1] = (char *) [argument UTF8String];
81  }
82 
83  // REGISTER THE ID BEFORE STARTING EXECUTION
84  executionId = newExecutionId;
85  addExecution(newExecutionId);
86 
87  // RUN
88  lastReturnCode = ffmpeg_execute(([arguments count] + 1), commandCharPArray);
89 
90  // ALWAYS REMOVE THE ID FROM THE MAP
91  removeExecution(newExecutionId);
92 
93  // CLEANUP
94  av_free(commandCharPArray[0]);
95  av_free(commandCharPArray);
96 
97  [executionsLock lock];
98  [executions removeObject: currentFFmpegExecution];
99  [executionsLock unlock];
100 
101  return lastReturnCode;
102 }

219 + (NSArray*)parseArguments:(NSString*)command {
220  NSMutableArray *argumentArray = [[NSMutableArray alloc] init];
221  NSMutableString *currentArgument = [[NSMutableString alloc] init];
222 
223  bool singleQuoteStarted = false;
224  bool doubleQuoteStarted = false;
225 
226  for (int i = 0; i < command.length; i++) {
227  unichar previousChar;
228  if (i > 0) {
229  previousChar = [command characterAtIndex:(i - 1)];
230  } else {
231  previousChar = 0;
232  }
233  unichar currentChar = [command characterAtIndex:i];
234 
235  if (currentChar == ' ') {
236  if (singleQuoteStarted || doubleQuoteStarted) {
237  [currentArgument appendFormat: @"%C", currentChar];
238  } else if ([currentArgument length] > 0) {
239  [argumentArray addObject: currentArgument];
240  currentArgument = [[NSMutableString alloc] init];
241  }
242  } else if (currentChar == '\'' && (previousChar == 0 || previousChar != '\\')) {
243  if (singleQuoteStarted) {
244  singleQuoteStarted = false;
245  } else if (doubleQuoteStarted) {
246  [currentArgument appendFormat: @"%C", currentChar];
247  } else {
248  singleQuoteStarted = true;
249  }
250  } else if (currentChar == '\"' && (previousChar == 0 || previousChar != '\\')) {
251  if (doubleQuoteStarted) {
252  doubleQuoteStarted = false;
253  } else if (singleQuoteStarted) {
254  [currentArgument appendFormat: @"%C", currentChar];
255  } else {
256  doubleQuoteStarted = true;
257  }
258  } else {
259  [currentArgument appendFormat: @"%C", currentChar];
260  }
261  }
262 
263  if ([currentArgument length] > 0) {
264  [argumentArray addObject: currentArgument];
265  }
266 
267  return argumentArray;
268 }
