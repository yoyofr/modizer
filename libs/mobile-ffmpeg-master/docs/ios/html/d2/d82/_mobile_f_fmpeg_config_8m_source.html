296 void mobileffmpeg_log_callback_function(void *ptr, int level, const char* format, va_list vargs) {
297 
298  // DO NOT PROCESS UNWANTED LOGS
299  if (level >= 0) {
300  level &= 0xff;
301  }
302  int activeLogLevel = [MobileFFmpegConfig getLogLevel];
303 
304  // AV_LOG_STDERR logs are always redirected
305  if ((activeLogLevel == AV_LOG_QUIET && level != AV_LOG_STDERR) || (level > activeLogLevel)) {
306  return;
307  }
308 
309  NSString *logData = [[NSString alloc] initWithFormat:[NSString stringWithCString:format encoding:NSUTF8StringEncoding] arguments:vargs];
310 
311  logCallbackDataAdd(level, logData);
312 
313  // BUILDING LAST COMMAND OUTPUT AS EARLY AS POSSIBLE
314  [lastCommandOutput appendString:logData];
315 }

335 void callbackBlockFunction() {
336  int activeLogLevel = av_log_get_level();
337  if ((activeLogLevel != AV_LOG_QUIET) && (AV_LOG_DEBUG <= activeLogLevel)) {
338  NSLog(@"Async callback block started.\n");
339  }
340 
341  while(redirectionEnabled) {
342  @autoreleasepool {
343  @try {
344 
345  CallbackData *callbackData = callbackDataRemove();
346  if (callbackData != nil) {
347 
348  if ([callbackData getType] == LogType) {
349 
350  // LOG CALLBACK
351  int activeLogLevel = [MobileFFmpegConfig getLogLevel];
352  int levelValue = [callbackData getLogLevel];
353 
354  if ((activeLogLevel == AV_LOG_QUIET && levelValue != AV_LOG_STDERR) || (levelValue > activeLogLevel)) {
355 
356  // LOG NEITHER PRINTED NOR FORWARDED
357  } else {
358  if (logDelegate != nil) {
359 
360  // FORWARD LOG TO DELEGATE
361  [logDelegate logCallback:[callbackData getExecutionId]:[callbackData getLogLevel]:[callbackData getLogData]];
362 
363  } else {
364  switch (levelValue) {
365  case AV_LOG_QUIET:
366  // PRINT NO OUTPUT
367  break;
368  default:
369  // WRITE TO NSLOG
370  NSLog(@"%@: %@", [MobileFFmpegConfig logLevelToString:[callbackData getLogLevel]], [callbackData getLogData]);
371  break;
372  }
373  }
374  }
375 
376  } else {
377 
378  // STATISTICS CALLBACK
379  Statistics *newStatistics = [[Statistics alloc] initWithId:[callbackData getExecutionId] videoFrameNumber:[callbackData getStatisticsFrameNumber] fps:[callbackData getStatisticsFps] quality:[callbackData getStatisticsQuality] size:[callbackData getStatisticsSize] time:[callbackData getStatisticsTime] bitrate:[callbackData getStatisticsBitrate] speed:[callbackData getStatisticsSpeed]];
380  [lastReceivedStatistics update:newStatistics];
381 
382  if (logDelegate != nil) {
383 
384  // FORWARD STATISTICS TO DELEGATE
385  [statisticsDelegate statisticsCallback:lastReceivedStatistics];
386  }
387  }
388 
389  } else {
390  callbackWait(100);
391  }
392 
393  } @catch(NSException *exception) {
394  activeLogLevel = av_log_get_level();
395  if ((activeLogLevel != AV_LOG_QUIET) && (AV_LOG_WARNING <= activeLogLevel)) {
396  NSLog(@"Async callback block received error: %@n\n", exception);
397  NSLog(@"%@", [exception callStackSymbols]);
398  }
399  }
400  }
401  }
402 
403  activeLogLevel = av_log_get_level();
404  if ((activeLogLevel != AV_LOG_QUIET) && (AV_LOG_DEBUG <= activeLogLevel)) {
405  NSLog(@"Async callback block stopped.\n");
406  }
407 }

615 + (void)setFontDirectory:(NSString*)fontDirectoryPath with:(NSDictionary*)fontNameMapping {
616  NSError *error = nil;
617  BOOL isDirectory = YES;
618  BOOL isFile = NO;
619  int validFontNameMappingCount = 0;
620  NSString *tempConfigurationDirectory = [NSTemporaryDirectory() stringByAppendingPathComponent:@".mobileffmpeg"];
621  NSString *fontConfigurationFile = [tempConfigurationDirectory stringByAppendingPathComponent:@"fonts.conf"];
622  int activeLogLevel = av_log_get_level();
623 
624  if (![[NSFileManager defaultManager] fileExistsAtPath:tempConfigurationDirectory isDirectory:&isDirectory]) {
625 
626  if (![[NSFileManager defaultManager] createDirectoryAtPath:tempConfigurationDirectory withIntermediateDirectories:YES attributes:nil error:&error]) {
627  if ((activeLogLevel != AV_LOG_QUIET) && (AV_LOG_WARNING <= activeLogLevel)) {
628  NSLog(@"Failed to set font directory. Error received while creating temp conf directory: %@.", error);
629  }
630  return;
631  }
632 
633  if ((activeLogLevel != AV_LOG_QUIET) && (AV_LOG_DEBUG <= activeLogLevel)) {
634  NSLog(@"Created temporary font conf directory: TRUE.");
635  }
636  }
637 
638  if ([[NSFileManager defaultManager] fileExistsAtPath:fontConfigurationFile isDirectory:&isFile]) {
639  BOOL fontConfigurationDeleted = [[NSFileManager defaultManager] removeItemAtPath:fontConfigurationFile error:NULL];
640  if ((activeLogLevel != AV_LOG_QUIET) && (AV_LOG_DEBUG <= activeLogLevel)) {
641  NSLog(@"Deleted old temporary font configuration: %s.", fontConfigurationDeleted?"TRUE":"FALSE");
642  }
643  }
644 
645  /* PROCESS MAPPINGS FIRST */
646  NSString *fontNameMappingBlock = @"";
647  for (NSString *fontName in [fontNameMapping allKeys]) {
648  NSString *mappedFontName = [fontNameMapping objectForKey:fontName];
649 
650  if ((fontName != nil) && (mappedFontName != nil) && ([fontName length] > 0) && ([mappedFontName length] > 0)) {
651 
652  fontNameMappingBlock = [NSString stringWithFormat:@"%@\n%@\n%@%@%@\n%@\n%@\n%@%@%@\n%@\n%@\n",
653  @" <match target=\"pattern\">",
654  @" <test qual=\"any\" name=\"family\">",
655  @" <string>", fontName, @"</string>",
656  @" </test>",
657  @" <edit name=\"family\" mode=\"assign\" binding=\"same\">",
658  @" <string>", mappedFontName, @"</string>",
659  @" </edit>",
660  @" </match>"];
661 
662  validFontNameMappingCount++;
663  }
664  }
665 
666  NSString *fontConfiguration = [NSString stringWithFormat:@"%@\n%@\n%@\n%@\n%@%@%@\n%@\n%@\n",
667  @"<?xml version=\"1.0\"?>",
668  @"<!DOCTYPE fontconfig SYSTEM \"fonts.dtd\">",
669  @"<fontconfig>",
670  @" <dir>.</dir>",
671  @" <dir>", fontDirectoryPath, @"</dir>",
672  fontNameMappingBlock,
673  @"</fontconfig>"];
674 
675  if (![fontConfiguration writeToFile:fontConfigurationFile atomically:YES encoding:NSUTF8StringEncoding error:&error]) {
676  if ((activeLogLevel != AV_LOG_QUIET) && (AV_LOG_WARNING <= activeLogLevel)) {
677  NSLog(@"Failed to set font directory. Error received while saving font configuration: %@.", error);
678  }
679  return;
680  }
681  if ((activeLogLevel != AV_LOG_QUIET) && (AV_LOG_DEBUG <= activeLogLevel)) {
682  NSLog(@"Saved new temporary font configuration with %d font name mappings.", validFontNameMappingCount);
683  }
684 
685  [MobileFFmpegConfig setFontconfigConfigurationPath:tempConfigurationDirectory];
686 
687  if ((activeLogLevel != AV_LOG_QUIET) && (AV_LOG_DEBUG <= activeLogLevel)) {
688  NSLog(@"Font directory %@ registered successfully.", fontDirectoryPath);
689  }
690 }
