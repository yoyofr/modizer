45 + (int)executeWithArguments:(NSArray*)arguments {
46  lastCommandOutput = [[NSMutableString alloc] init];
47 
48  char **commandCharPArray = (char **)av_malloc(sizeof(char*) * ([arguments count] + 1));
49 
50  /* PRESERVING CALLING FORMAT
51  *
52  * ffprobe <arguments>
53  */
54  commandCharPArray[0] = (char *)av_malloc(sizeof(char) * ([LIB_NAME length] + 1));
55  strcpy(commandCharPArray[0], [LIB_NAME UTF8String]);
56 
57  for (int i=0; i < [arguments count]; i++) {
58  NSString *argument = [arguments objectAtIndex:i];
59  commandCharPArray[i + 1] = (char *) [argument UTF8String];
60  }
61 
62  // RUN
63  lastReturnCode = ffprobe_execute(([arguments count] + 1), commandCharPArray);
64 
65  // CLEANUP
66  av_free(commandCharPArray[0]);
67  av_free(commandCharPArray);
68 
69  return lastReturnCode;
70 }
